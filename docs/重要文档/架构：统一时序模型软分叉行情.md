utils/timeframe_selection.py
import pandas as pd
import numpy as np
from statsmodels.tsa.stattools import acf

def estimate_effective_horizon(returns: pd.Series, max_lag=50):
"""估计收益自相关显著的最远 horizon"""
acf_vals = acf(returns.dropna(), nlags=max_lag, fft=True)
# 找到第一个不显著的 lag（95% 置信区间）
conf_int = 1.96 / np.sqrt(len(returns))
for i in range(1, len(acf_vals)):
if abs(acf_vals[i]) < conf_int:
return max(1, i - 1)
return max_lag

def find_optimal_timeframe(df: pd.DataFrame, candidate_tfs=['5T', '15T', '1H', '4H']):
"""通过重采样 + IC 衰减选择最优 timeframe"""
best_tf, best_ic = None, -1
raw_returns = df['close'].pct_change()

for tf in candidate_tfs:
resampled = df.resample(tf).last().dropna()
if len(resampled) < 50:
continue
returns = resampled['close'].pct_change()
horizon = estimate_effective_horizon(returns)
if horizon < 1:
continue

# 构建简单动量因子测试 IC
mom = returns.rolling(20).mean().shift(1)
ic = mom.corr(returns.shift(-horizon))
if ic > best_ic:
best_ic = ic
best_tf = tf
return best_tf or '1H', max(1, estimate_effective_horizon(raw_returns))

)

# 5. 添加 timeframe 标识（用于后续归一化分组）

return engineered

scaler = RobustScaler()

scalers[tf_group] = scaler
else:
scaler = scalers[tf_group]
X_norm.loc[mask, cols] = scaler.transform(X_norm.loc[mask, cols])

if fit and scaler_path:
with open(scaler_path, 'wb') as f:

import lightgbm as lgb
from sklearn.model_selection import TimeSeriesSplit

class PurgedWalkForwardCV:
def __init__(self, n_splits=5, purge_gap=10, embargo_gap=5):
self.n_splits = n_splits
self.purge_gap = purge_gap
self.embargo_gap = embargo_gap

def split(self, X, y=None, groups=None):
tscv = TimeSeriesSplit(n_splits=self.n_splits)
indices = np.arange(len(X))
for train_idx, val_idx in tscv.split(X):
# Purge: 删除训练集末尾 gap 天
train_end = train_idx[-1]
purge_start = train_end - self.purge_gap + 1
train_idx = train_idx[train_idx < purge_start]

# Embargo: 删除验证集开头 gap 天
val_start = val_idx[0]
embargo_end = val_start + self.embargo_gap
val_idx = val_idx[val_idx >= embargo_end]

if len(train_idx) > 0 and len(val_idx) > 0:
yield train_idx, val_idx

def train_unified_model(X, y, meta):
cv = PurgedWalkForwardCV(n_splits=4, purge_gap=20, embargo_gap=10)
params = {
'objective': 'regression',
'metric': 'rmse',
'learning_rate': 0.01,
'num_leaves': 64,
'feature_fraction': 0.8,
'bagging_fraction': 0.8,
'verbose': -1,
'seed': 42
}

models = []
for fold, (train_idx, val_idx) in enumerate(cv.split(X)):
dtrain = lgb.Dataset(X.iloc[train_idx], y.iloc[train_idx])
dval = lgb.Dataset(X.iloc[val_idx], y.iloc[val_idx])

model = lgb.train(
params,
dtrain,
valid_sets=[dval],
num_boost_round=1000,
callbacks=[lgb.early_stopping(50)]
)
models.append(model)
print(f"Fold {fold} done.")

# 返回平均模型或最佳模型
return models[-1] # 或 ensemble

| 问题       | 传统做法       | 该方案解决方式                  |
| -------- | ---------- | ------------------------ |

## 🧮 四、优缺点总结

| 方面        | 优点                                  | 缺点 / 风险                               |
| --------- | ----------------------------------- | ------------------------------------- |
| **泛化能力**  | 统一模型可学习不同币的共性结构（趋势、波动）              | 若币种差异过大（如BTC vs MEME币），模型可能平均化，丢失局部特征 |
| **样本利用**  | 合并样本增加训练量，防止过拟合                     | 不同时间尺度的数据需归一化处理，否则引入噪声                |
| **工程复杂度** | 单模型部署、特征模块化，实盘简单                    | 前期特征生成管线复杂（需按币配置）                     |
| **可解释性**  | SHAP/Feature Importance 可追踪每个币的特征影响 | 模型内部逻辑仍属黑箱，部分风控难解释                    |
| **更新与维护** | 易于更新模型，只需重训一份                       | 若市场 regime 变化极大，仍需定期重训（如季度）           |
| **实盘鲁棒性** | 统一规则便于风控与仓位分配                       | 对小众币种可能信号衰减，需额外噪声过滤                   |

# 再训练一个hmm还不如把这些放统一模型吧

对，你的想法很合理——在很多生产级量化系统里，确实可以**把 Regime Detect 信号也作为特征输入到统一 LightGBM 模型里，而不单独训练 HMM**。这样做有几个优点和注意事项：

---

### 1️⃣ **为什么可以直接放到统一模型**

* **状态作为特征**：

  * 不管是 rule-based 状态还是 HMM 状态，都可以作为额外特征加入 `X`
  * LightGBM 会自动学习状态对信号的加权作用：

    * `is_trending × rsi`
    * `vol_regime × bb_width`
* **统一训练**：

  * 所有币种数据共用一个模型
  * 模型同时学习特征与状态之间的交互
* **避免额外模型维护**：

  * 不需要单独训练 HMM
  * 实盘只需一个模型即可输出预测信号和仓位建议

---

### 2️⃣ **优缺点分析**

| 方法                        | 优点                       | 缺点                                  |
| ------------------------- | ------------------------ | ----------------------------------- |
| 单独 HMM + LightGBM         | 状态平滑、解释性好                | 增加模型复杂度和维护成本；HMM 的训练窗口、状态数需要调参      |
| Rule/HMM 直接做特征输入 LightGBM | 简化架构；模型自适应不同市场状态；可以跨币种共享 | 可能略丢失 HMM 平滑特性，需要设计合理的状态特征（如滑动窗口均值） |

---

### 3️⃣ **实践建议**

1. **用 rule 或简单指标生成 Regime 特征**

   * `is_trending`, `is_compressed`, `vol_regime`
   * 可加滑动均值或 EWMA 平滑
2. **作为特征加入统一 LightGBM 模型**

   * 可以和币种标识、短中长周期特征交互
3. **模型自动学习不同状态下的信号权重**

   * 模型会自己发现：在趋势状态下 rsi 重要，在压缩状态下 bb_width 重要
4. **可选**：初期可做一个 HMM 版本验证效果，然后用 rule 特征替代 HMM 输出，简化生产架构

---

✅ **总结**：

> 如果目标是生产化、跨币种、可维护性高 → **统一模型 + 状态特征输入** 是最稳妥方案
> 不必额外训练 HMM，除非你对状态平滑和解释性有特别需求
