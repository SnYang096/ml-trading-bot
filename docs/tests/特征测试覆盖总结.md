### 1. 未来数据泄露测试

```python
def test_no_future_leak(self):
    """验证特征不使用未来数据"""
    # 在时间点 t 制造突变
    original_value = df.loc[t, "close"]
    df.loc[t, "close"] = original_value * 1.5
    
    result = extract_features(df, window=w)
    
    # t 时刻的特征应该只用到 t-w 到 t-1 的数据
    # 不应该包含 t 时刻的突变
    feature_t = result.loc[t, "feature_name"]
    feature_t_plus_1 = result.loc[t+1, "feature_name"]
    
    # 验证 t 时刻的特征不包含 t 时刻的数据
    self.assertNotEqual(feature_t, feature_t_plus_1)
```

### 2. 多资产归一化测试

```python
def test_multi_asset_normalization(self):
    """验证多资产归一化"""
    # 创建不同价格水平的资产
    df_btc = create_test_data(price_level=50000)
    df_eth = create_test_data(price_level=3000)
    
    # 合并并添加 _symbol 列
    df_multi = pd.concat([df_btc, df_eth])
    df_multi["_symbol"] = ["BTC"] * len(df_btc) + ["ETH"] * len(df_eth)
    
    # 计算特征（应该自动归一化）
    result = extract_features(df_multi, group_col="_symbol")
    
    # 验证不同资产的分布相似
    btc_features = result[result["_symbol"] == "BTC"]["feature_name"]
    eth_features = result[result["_symbol"] == "ETH"]["feature_name"]
    
    # 均值应该接近（归一化后）
    self.assertAlmostEqual(btc_features.mean(), eth_features.mean(), places=1)
```

### 3. 模拟数据测试

```python
def test_with_simulated_data(self):
    """使用模拟数据测试特征"""
    # 创建模拟数据
    np.random.seed(42)
    n_samples = 1000
    prices = 100 + np.cumsum(np.random.randn(n_samples) * 0.5)
    
    df = pd.DataFrame({
        "close": prices,
        "high": prices * 1.01,
        "low": prices * 0.99,
        "volume": np.random.uniform(1000, 10000, n_samples),
    })
    
    # 计算特征
    result = extract_features(df)
    
    # 验证特征的有效性
    self.assertFalse(result.isna().all().any())
    self.assertGreater(result["feature_name"].std(), 0)
```
