### 以 WPT entropy 为例（其他同理）

```python
wpt_rel = (wpt_short_mean - wpt_long_mean) / wpt_long_std
```

```python
wpt_score = clip(sigmoid(wpt_rel), 0, 1)
```

解释一下语义：

* `≈0.5`：正常环境
* `→1`：比近期明显更吵
* `→0`：结构反而更清晰（少见，但好事）

```text
wpt_score
```

### **修改1：添加Noise Penalty计算模块**

```python
# src/time_series_model/execution/noise_penalty.py

class ExecutionNoisePenalty:
    """执行噪声惩罚计算器"""
    
    def __init__(self, config: dict):
        self.short_window = config.get('short_window', 20)
        self.long_window = config.get('long_window', 120)
        self.weights = config.get('weights', {
            'wpt': 0.35,
            'spectrum': 0.30,
            'hilbert': 0.20,
            'hurst': 0.15
        })
    
    def compute(self, df: pd.DataFrame) -> pd.Series:
        """
        计算execution_noise_penalty
        
        Returns:
            0-0.8之间的连续值（永远不到1，避免完全deny）
        """
        # 1. 每个特征的相对化
        wpt_rel = self._relativize(df['wpt_entropy'], self.short_window, self.long_window)
        spectrum_rel = self._relativize(df['spectrum_width'], self.short_window, self.long_window)
        hilbert_rel = self._relativize(df['hilbert_phase_var'], self.short_window, self.long_window)
        hurst_rel = self._relativize(df['hurst'], self.short_window, self.long_window, inverse=True)
        
        # 2. 转为0-1评分
        wpt_score = self._sigmoid(wpt_rel)
        spectrum_score = self._sigmoid(spectrum_rel)
        hilbert_score = self._sigmoid(hilbert_rel)
        hurst_score = self._sigmoid(hurst_rel)
        
        # 3. 加权合成
        noise_raw = (
            self.weights['wpt'] * wpt_score +
            self.weights['spectrum'] * spectrum_score +
            self.weights['hilbert'] * hilbert_score +
            self.weights['hurst'] * hurst_score
        )
        
        # 4. 钳制到0-0.8（永远不到1）
        noise_penalty = np.clip(noise_raw, 0.0, 0.8)
        
        # 5. 惯性滤波（防抖动）
        noise_penalty_smooth = noise_penalty.ewm(alpha=0.3).mean()
        
        return noise_penalty_smooth
    
    def _relativize(self, series, short_win, long_win, inverse=False):
        """相对化：(短期均值 - 长期均值) / 长期标准差"""
        short_mean = series.rolling(short_win).mean()
        long_mean = series.rolling(long_win).mean()
        long_std = series.rolling(long_win).std()
        
        rel = (short_mean - long_mean) / (long_std + 1e-8)
        
        if inverse:
            rel = -rel  # Hurst是反向的
        
        return rel
    
    def _sigmoid(self, x):
        """Sigmoid转换到0-1"""
        return 1 / (1 + np.exp(-x))
```

### **修改2：Execution Tier消费Noise Penalty**

```python
# src/time_series_model/execution/tier.py

@dataclass
class ExecutionParams:
    """执行参数"""
    sl_r: float
    tp_r: float
    size_multiplier: float
    
    # 噪声调整系数
    noise_adjustment: Optional[dict] = None
    
    def adjust_for_noise(self, noise_penalty: float) -> 'ExecutionParams':
        """根据噪声惩罚调整参数"""
        if self.noise_adjustment is None:
            return self  # 不调整
        
        adjusted = ExecutionParams(
            sl_r=self.sl_r * (1 + self.noise_adjustment['sl_factor'] * noise_penalty),
            tp_r=self.tp_r * (1 - self.noise_adjustment['tp_factor'] * noise_penalty),
            size_multiplier=self.size_multiplier * (1 - self.noise_adjustment['size_factor'] * noise_penalty),
        )
        
        return adjusted
```
