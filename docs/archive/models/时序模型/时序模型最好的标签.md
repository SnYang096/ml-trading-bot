🔧 第一步：生成高质量回归目标 + 评估标签

def prepare_time_series_target(
df: pd.DataFrame,
price_col: str = "close",
asset_col: str = "symbol", # 设为 None 表示单资产
date_col: str = "date",
hold_period: int = 5,
lookback_window: int = 60,
vol_mult: float = 0.5,
min_samples: int = 30
):
"""
返回：
X: 特征（需用户自行提供，此处只处理目标）
y_reg: 回归目标（波动率标准化后的未来收益）
eval_labels: 用于 Rank IC 和信号生成的辅助列
"""
df = df.copy()
df[date_col] = pd.to_datetime(df[date_col])
group_cols = [asset_col] if asset_col else []

# 1. 基础收益计算
df["future_return"] = df.groupby(group_cols)[price_col].pct_change(hold_period).shift(-hold_period)

# 2. 滚动波动率（基于 hold_period 收益）
def _rolling_vol(series):
rets = series.pct_change(hold_period).dropna()
return rets.rolling(window=lookback_window, min_periods=min_samples).std()

df["rolling_vol"] = df.groupby(group_cols)[price_col].transform(_rolling_vol)

# 3. 波动率标准化目标（Sharpe-like）
df["target"] = df["future_return"] / (df["rolling_vol"] + 1e-8)

# 4. 计算未来收益在其历史分布中的分位数（用于评估和信号）
def _compute_quantile(group):
quantiles = []
returns = group["future_return"]
for i in range(len(group)):
if i < lookback_window + hold_period:
quantiles.append(np.nan)
continue
hist_rets = returns.iloc[i - lookback_window - hold_period : i - hold_period].dropna()
if len(hist_rets) < min_samples:
quantiles.append(np.nan)
continue
current = returns.iloc[i]
if pd.isna(current):
quantiles.append(np.nan)
continue
q = (hist_rets < current).mean()
quantiles.append(q)
return pd.Series(quantiles, index=group.index)

if asset_col:
df["return_quantile"] = df.groupby(asset_col).apply(_compute_quantile).droplevel(0)
else:
df["return_quantile"] = _compute_quantile(df)

# 5. 可交易信号掩码（用于过滤低质量样本）
df["tradable"] = (
df["future_return"].abs() > (vol_mult * df["rolling_vol"])
) & df["return_quantile"].between(0.1, 0.9, inclusive="neither") # 排除极端尾部噪声

return df

🧠 第二步：训练以 Rank IC 为导向的回归模型

def train_rank_ic_optimized_model(
df: pd.DataFrame,
feature_cols: list,
target_col: str = "target",
tradable_mask_col: str = "tradable",
n_splits: int = 5,
hold_period: int = 5,
lgbm_params
):
"""
使用时间序列 CV 训练模型，每折记录 Rank IC
"""
df = df.dropna(subset=feature_cols + [target_col, "return_quantile"]).copy()
df = df.sort_values("date").reset_index(drop=True)

tscv = TimeSeriesSplit(n_splits=n_splits)
models = []
ic_scores = []

X = df[feature_cols]
y = df[target_col]
y_true_return = df["future_return"] # 用于计算真实 Rank IC
tradable = df[tradable_mask_col]

for fold, (train_idx, val_idx) in enumerate(tscv.split(X)):
# 时间对齐：确保验证集在训练集之后
train_end_date = df.loc[train_idx, "date"].max()
val_start_date = df.loc[val_idx, "date"].min()
assert val_start_date > train_end_date, "Time leakage!"

# 只用可交易样本训练（可选）
train_mask = tradable.iloc[train_idx] & (df.loc[train_idx, "return_quantile"].notna())
X_train = X.iloc[train_idx][train_mask]
y_train = y.iloc[train_idx][train_mask]

X_val = X.iloc[val_idx]
y_val_true_ret = y_true_return.iloc[val_idx]
quantile_val = df.loc[val_idx, "return_quantile"]

# 训练
model = LGBMRegressor(
objective='regression',
metric='mse',
boosting_type='gbdt',
n_estimators=500,
learning_rate=0.05,
max_depth=6,
subsample=0.8,
colsample_bytree=0.8,
random_state=42,
lgbm_params
)
model.fit(X_train, y_train, verbose=False)

# 预测 & 评估 Rank IC
pred = model.predict(X_val)
valid_mask = quantile_val.notna()
if valid_mask.sum() > 10:
ic = pd.Series(pred[valid_mask]).corr(
pd.Series(y_val_true_ret[valid_mask]), method='spearman'
)
ic_scores.append(ic if not pd.isna(ic) else 0)
print(f"Fold {fold+1}: Rank IC = {ic:.4f}")
else:
ic_scores.append(0)

models.append(model)

print(f"\n✅ Avg Rank IC: {np.mean(ic_scores):.4f} ± {np.std(ic_scores):.4f}")
return models, np.mean(ic_scores)

🚀 第三步：生成动态交易信号（Long/Short/Hold）

def generate_trading_signals(df, models, feature_cols, quantile_threshold=0.9):
"""
使用模型预测，并基于预测值的历史分位数生成信号
"""
df = df.copy()
X = df[feature_cols]

# 集成预测（多模型平均）
preds = np.mean([model.predict(X) for model in models], axis=0)
df["pred"] = preds

# 对每个资产，计算预测值在其近期预测分布中的分位数
def _pred_quantile(group):
return group["pred"].rolling(window=30, min_periods=10).rank(pct=True)

if "symbol" in df.columns:
df["pred_quantile"] = df.groupby("symbol").apply(_pred_quantile).droplevel(0)
else:
df["pred_quantile"] = _pred_quantile(df)

# 动态信号
df["signal"] = 0
df.loc[df["pred_quantile"] > quantile_threshold, "signal"] = 1 # Long
df.loc[df["pred_quantile"] < (1 - quantile_threshold), "signal"] = -1 # Short

# 可选：叠加波动率过滤
# df.loc[df["rolling_vol"] < df["rolling_vol"].quantile(0.2), "signal"] = 0

return df

import pandas as pd
import numpy as np
from lightgbm import LGBMRegressor
from sklearn.model_selection import TimeSeriesSplit

def prepare_time_series_target(
df: pd.DataFrame,
price_col: str = "close",
asset_col: str = "symbol",
date_col: str = "date",
hold_period: int = 5,
lookback_window: int = 60,
vol_mult: float = 0.5,
min_samples: int = 30
):
"""
新增输出：
'sample_weight': 用于训练时加权（趋势越强，权重越高）
"""
df = df.copy()
df[date_col] = pd.to_datetime(df[date_col])
group_cols = [asset_col] if asset_col else []

# 1. 基础收益
df["future_return"] = df.groupby(group_cols)[price_col].pct_change(hold_period).shift(-hold_period)

# 2. 滚动波动率
def _rolling_vol(series):
rets = series.pct_change(hold_period).dropna()
return rets.rolling(window=lookback_window, min_periods=min_samples).std()

df["rolling_vol"] = df.groupby(group_cols)[price_col].transform(_rolling_vol)

# 3. 波动率标准化目标
df["target"] = df["future_return"] / (df["rolling_vol"] + 1e-8)

# 4. 历史分位数标签（用于评估）
def _compute_quantile(group):
quantiles = []
returns = group["future_return"]
for i in range(len(group)):
if i < lookback_window + hold_period:
quantiles.append(np.nan)
continue
hist_rets = returns.iloc[i - lookback_window - hold_period : i - hold_period].dropna()
if len(hist_rets) < min_samples:
quantiles.append(np.nan)
continue
current = returns.iloc[i]
if pd.isna(current):
quantiles.append(np.nan)
continue
q = (hist_rets < current).mean()
quantiles.append(q)
return pd.Series(quantiles, index=group.index)

if asset_col:
df["return_quantile"] = df.groupby(asset_col).apply(_compute_quantile).droplevel(0)
else:
df["return_quantile"] = _compute_quantile(df)

# 5. 样本权重：趋势强度 = momentum / vol （需用户提供 momentum 特征）
# 如果没有 momentum，可用简单移动平均斜率替代
if "momentum" not in df.columns:
# 简单替代：20日均线斜率
ma20 = df.groupby(group_cols)[price_col].transform(lambda x: x.rolling(20).mean())
df["momentum"] = df.groupby(group_cols)["ma20"].diff(5) / df[price_col]

df["trend_strength"] = (df["momentum"].abs() / (df["rolling_vol"] + 1e-8)).clip(lower=0.1, upper=5.0)
df["sample_weight"] = df["trend_strength"]

return df

def train_rank_ic_optimized_model(
df: pd.DataFrame,
feature_cols: list,
target_col: str = "target",
weight_col: str = "sample_weight",
n_splits: int = 5,
hold_period: int = 5,
lgbm_params
):
df = df.dropna(subset=feature_cols + [target_col, "return_quantile", weight_col]).copy()
df = df.sort_values("date").reset_index(drop=True)

tscv = TimeSeriesSplit(n_splits=n_splits)
models = []
ic_scores = []

X = df[feature_cols]
y = df[target_col]
y_true_return = df["future_return"]
weights = df[weight_col]

for fold, (train_idx, val_idx) in enumerate(tscv.split(X)):
train_end_date = df.loc[train_idx, "date"].max()
val_start_date = df.loc[val_idx, "date"].min()
assert val_start_date > train_end_date, "Time leakage!"

X_train, y_train, w_train = X.iloc[train_idx], y.iloc[train_idx], weights.iloc[train_idx]
X_val, y_val_true_ret = X.iloc[val_idx], y_true_return.iloc[val_idx]
quantile_val = df.loc[val_idx, "return_quantile"]

model = LGBMRegressor(
objective='regression',
metric='mse',
boosting_type='gbdt',
n_estimators=500,
learning_rate=0.05,
max_depth=6,
subsample=0.8,
colsample_bytree=0.8,
random_state=42,
lgbm_params
)
model.fit(X_train, y_train, sample_weight=w_train, verbose=False)

pred = model.predict(X_val)
valid_mask = quantile_val.notna()
if valid_mask.sum() > 10:
ic = pd.Series(pred[valid_mask]).corr(
pd.Series(y_val_true_ret[valid_mask]), method='spearman'
)
ic_scores.append(ic if not pd.isna(ic) else 0)
print(f"Fold {fold+1}: Rank IC = {ic:.4f}")
else:
ic_scores.append(0)

models.append(model)

print(f"\n✅ Avg Rank IC: {np.mean(ic_scores):.4f} ± {np.std(ic_scores):.4f}")
return models, np.mean(ic_scores)

def generate_trading_signals(
df: pd.DataFrame,
models: list,
feature_cols: list,
confidence_threshold: float = 0.85, # 置信度阈值（分位数距离中性区）
min_pred_quantile_for_long: float = 0.9,
min_pred_quantile_for_short: float = 0.1
):
"""
新增：
confidence_score = pred_quantile - 0.5 2 → [0, 1]
只有 confidence_score >= confidence_threshold 才交易
"""
df = df.copy()
X = df[feature_cols]

preds = np.mean([model.predict(X) for model in models], axis=0)
df["pred"] = preds

# 预测值的历史分位数（滚动30天）
def _pred_quantile(group):
return group["pred"].rolling(window=30, min_periods=10).rank(pct=True)

if "symbol" in df.columns:
df["pred_quantile"] = df.groupby("symbol").apply(_pred_quantile).droplevel(0)
else:
df["pred_quantile"] = _pred_quantile(df)

# 置信度 = 距离中性区（0.5）的绝对距离，映射到 [0, 1]
df["confidence_score"] = (df["pred_quantile"] - 0.5).abs() 2

# 初始化信号
df["signal"] = 0

# 高置信 Long
long_cond = (
(df["pred_quantile"] >= min_pred_quantile_for_long) &
(df["confidence_score"] >= confidence_threshold)
)
df.loc[long_cond, "signal"] = 1

# 高置信 Short
short_cond = (
(df["pred_quantile"] <= min_pred_quantile_for_short) &
(df["confidence_score"] >= confidence_threshold)
)
df.loc[short_cond, "signal"] = -1

return df
