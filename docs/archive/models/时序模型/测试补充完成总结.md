streaming_results = []

    df_stream = df.iloc[: i + 1].copy()
    stream_result = extract_features(df_stream)
    streaming_results.append(stream_result.iloc[-1])

batch_vals = batch_result["feature_col"].iloc[window:].dropna()

def test_lag_effect_decay_smoothly():
    """测试 lag 特征有效性应缓慢衰减"""
    features = extract_features(df)
    
    # 构造目标变量（仅用于测试）
    log_ret = np.log(df["close"] / df["close"].shift(1))
    target = log_ret.rolling(5).std().shift(-5)
    
    # 检查 lag 特征与目标的相关性
    lag_cols = ["feature_lag_1", "feature_lag_2", "feature_lag_3"]
    correlations = []
    for col in lag_cols:
        if col in features.columns:
            corr = features[col].corr(target, min_periods=50)
            correlations.append(abs(corr))
    
    # 应大致递减，且无断崖
    if len(correlations) >= 2:
        assert correlations[1] > 0.1 * correlations[0], "lag_2 相关性断崖式下降"
