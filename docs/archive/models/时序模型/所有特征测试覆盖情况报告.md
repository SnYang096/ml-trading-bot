def test_<feature>_no_future_leak(self, sample_data):
    """测试：无未来函数（修改未来数据不影响历史特征值）"""
    df = sample_data.copy()
    
    # 计算第一次特征
    result1 = extract_features(df)
    feature_1 = result1["feature_col"].copy()
    
    # 修改未来数据
    df_future_modified = df.copy()
    if len(df) > 100:
        df_future_modified.loc[df_future_modified.index[100]:, "close"] *= 2.0
        
        # 重新计算特征
        result2 = extract_features(df_future_modified)
        feature_2 = result2["feature_col"].copy()
        
        # 检查前50个时间点的特征值
        check_idx = df.index[:50]
        diff = (feature_1.loc[check_idx] - feature_2.loc[check_idx]).abs()
        assert diff.max() < 1e-6, "未来数据变化不应影响历史特征值"

### 流式vs批量一致性测试模板

```python
def test_<feature>_streaming_vs_batch_consistency(self, sample_data):
    """测试：流式 vs 批量一致性"""
    df = sample_data.copy()
    window = 60
    
    # 批量计算
    batch_result = extract_features(df)
    batch_vals = batch_result["feature_col"].iloc[window:].dropna()
    
    # 流式计算（分块处理）
    streaming_results = []
    for i in range(window, len(df)):
        df_stream = df.iloc[: i + 1].copy()
        stream_result = extract_features(df_stream)
        streaming_results.append(stream_result.iloc[-1])
    
    streaming_df = pd.DataFrame(streaming_results, index=df.index[window:])
    stream_vals = streaming_df["feature_col"].dropna()
    
    # 比较结果
    common_idx = batch_vals.index.intersection(stream_vals.index)
    diff = (batch_vals.loc[common_idx] - stream_vals.loc[common_idx]).abs()
    
    assert diff.max() < 1e-5, "流式与批量计算不一致"
```
