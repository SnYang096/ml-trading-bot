# 实施文档 01：2024 牛市 5x 趋势骑乘（精简版）

## 1. 目标

- 在 2024 牛市窗口验证 `bpc-long-120T`、`me-long-120T` 是否能稳定接近 `5x` 杠杆。
- 验证“先保本再加仓”的风险约束仍成立。
- 验证收益主要来自趋势段，而不是短噪音。

python - <<'PY'
from src.feature_store.feature_store import FeatureStore, FeatureStoreSpec

root = "feature_store"
layer = "features_bpc-short-120T_120T_81b6837b97"
timeframe = "120T"
month = "2024-09"
symbols = ["BTCUSDT","ETHUSDT","BNBUSDT","SOLUSDT","XRPUSDT","ADAUSDT"]

drop_cols = {
    "fp_poc","fp_vah","fp_val","fp_hvn","fp_lvn","fp_max_imbalance_price","fp_exhaustion_price"
}
base_cols = ["open","high","low","close","volume","_symbol","symbol"]

store = FeatureStore(root)
for sym in symbols:
    spec = FeatureStoreSpec(layer=layer, symbol=sym, timeframe=timeframe)
    df = store.read_month(spec, month)
    meta = store.read_month_meta(spec, month)
    keep = [c for c in df.columns if c not in drop_cols]
    df2 = df[keep].copy()
    feat_cols = [c for c in df2.columns if c not in base_cols]
    store.write_month(
        spec, month, df2,
        base_columns=base_cols,
        feature_columns=feat_cols,
        overwrite=True,
        metadata=(meta.get("metadata", {}) if isinstance(meta, dict) else {}),
    )
    print("patched", sym, month)
PY
