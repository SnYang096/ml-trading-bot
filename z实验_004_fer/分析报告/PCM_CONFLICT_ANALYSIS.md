### 🔍 冲突处理逻辑伪代码

```python
for each bar:
    signals = []
    
    # 1. 收集所有策略信号
    if fer_has_signal:
        signals.append(('fer', fer_signal))
    if me_has_signal:
        signals.append(('me', me_signal))
    if bpc_has_signal:
        signals.append(('bpc', bpc_signal))
    
    # 2. 按优先级仲裁
    if len(signals) > 1:  # 有冲突
        conflict_count += 1
        # 固定优先级: FER > ME > BPC
        if 'fer' in signals:
            chosen = fer_signal
            rejected = [s for s in signals if s != 'fer']
        elif 'me' in signals:
            chosen = me_signal
            rejected = [s for s in signals if s != 'me']
        else:
            chosen = bpc_signal
            rejected = []
    else:  # 无冲突
        chosen = signals[0] if signals else None
        rejected = []
    
    # 3. 执行选中信号
    if chosen:
        execute_trade(chosen)
    
    # 4. 记录被拒信号（用于反事实分析）
    for signal in rejected:
        record_counterfactual(signal)
```
